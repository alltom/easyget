      def serve_rails(env)
        request         = Request.new(env)
        response        = Response.new
        
        session_options = ActionController::CgiRequest::DEFAULT_SESSION_OPTIONS
        cgi             = CGIWrapper.new(request, response)
    
        Dispatcher.dispatch(cgi, session_options, response)

        response.finish
      end
